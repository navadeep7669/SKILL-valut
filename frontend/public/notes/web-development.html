/* Auto-responsive grid — no media queries needed! */
.card-grid {
  display: grid;
  grid-template-columns: repeat(auto-fit, minmax(250px, 1fr));
  gap: 1.5rem;
  padding: 1rem;
}

/* Perfect centering with Flexbox */
.center-me {
  display: flex;
  justify-content: center;   /* horizontal axis */
  align-items: center;       /* vertical axis */
  min-height: 100vh;
}

/* Responsive nav with Flexbox */
.navbar {
  display: flex;

}

JavaScript — Modern Async + Array Methods
// ✅ Clean async/await with full error handling
async function fetchUser(userId) {
  try {
    const res = await fetch(`/api/users/${userId}`);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (err) {
    console.error('Failed:', err.message);
    return null;  // always return a safe fallback!
  }
}

// ✅ Array methods — filter + map + sort chaining
const topUsers = users
  .filter(u => u.score >= 80)          // keep matching
  .map(u => ({ ...u, grade: 'A' }))    // transform each
  .sort((a, b) => b.score - a.score);  // sort descending

// ✅ Destructuring + default values
const { name = 'Anonymous', email = '' } = user ?? {};
const [first, second, ...rest] = sortedItems;
💡 Rule: Use const by default. Use let only when reassigning. Never use var — it has confusing function-scoped behaviour and hoisting bugs.

React — useEffect + useState Pattern
import { useState, useEffect } from 'react';

function UserProfile({ userId }) {
  const [user, setUser]       = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError]     = useState(null);

  useEffect(() => {
    setLoading(true);
    setError(null);

    const controller = new AbortController();

    fetch(`/api/users/${userId}`, { signal: controller.signal })
      .then(res => { if (!res.ok) throw new Error('Not found'); return res.json(); })
      .then(data => { setUser(data); setLoading(false); })
      .catch(err => { if (err.name !== 'AbortError') setError(err.message); setLoading(false); });

    // Cleanup: cancel fetch if userId changes before it completes
    return () => controller.abort();
  }, [userId]);  // ← re-runs whenever userId changes

  if (loading) return <div>Loading...</div>;
  if (error)   return <div>Error: {error}</div>;
  return <div>{user?.name}</div>;
}

Hooks (useState etc.)❌ Not available✅ Full React hooks access

Next.js App Router — Server Component (app/page.tsx)
// No "use client" = Server Component
// Can await data directly — no useEffect needed!
async function ProductsPage() {
  const products = await fetch('https://api.example.com/products', {
    next: { revalidate: 60 }  // ISR: refresh cache every 60s
  }).then(r => r.json());

  return (
    <main>
      <h1>Products</h1>
      <ul>
        {products.map(p => (
          <li key={p.id}>{p.name} — ${p.price}</li>
        ))}
      </ul>
    </main>
  );
}
⚠️ Common Mistake: Don't add "use client" to every file. Keep components as Server Components by default — they're faster and more secure. Only add "use client" when you need interactivity (onClick, useState, etc.).

Node.js — Express REST API (Complete Example)
const express = require('express');
const app = express();
app.use(express.json());

let products = [{ id: 1, name: 'Widget', price: 9.99 }];

// GET — list all
app.get('/api/products', (req, res) => {
  res.json({ success: true, data: products, count: products.length });
});

// GET — single item
app.get('/api/products/:id', (req, res) => {
  const product = products.find(p => p.id === +req.params.id);
  if (!product) return res.status(404).json({ error: 'Not found' });
  res.json({ success: true, data: product });
});

// POST — create
app.post('/api/products', (req, res) => {
  const { name, price } = req.body;
  if (!name || price == null) {
    return res.status(400).json({ error: 'name and price required' });
  }
  const product = { id: Date.now(), name, price: +price };
  products.push(product);
  res.status(201).json({ success: true, data: product });
});

// Global error handler
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({ error: 'Internal Server Error' });
});

app.listen(3001, () => console.log('✅ API running on :3001'));
SQL vs NoSQL — Decision Guide
AspectSQL (PostgreSQL)NoSQL (MongoDB)
Data structureFixed schema, tables & rowsFlexible JSON documents
Best forFinancial, relational, reportsProfiles, logs, flexible schemas
ScalingVertical (bigger server)Horizontal (more servers)
QueriesSELECT * FROM users WHERE id=1db.users.findOne({id:1})
Joins✅ Native JOINs with foreign keys❌ Manual or aggregation pipeline
💡 Tip: Start with PostgreSQL for most apps — it handles structured data AND JSON columns (JSONB). Only switch to MongoDB if your data is truly schema-less from day one.

GitHub Actions — CI/CD Pipeline
YAML — .github/workflows/deploy.yml
name: Test & Deploy

on:
  push:
    branches: [main]        # triggers on every push to main
  pull_request:
    branches: [main]        # also runs on PRs for safety

jobs:
  test-and-deploy:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js 20
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'       # cache node_modules for speed

      - run: npm ci          # clean install (faster + reproducible)
      - run: npm test        # ← deploy is blocked if tests fail
      - run: npm run build

      - name: Deploy to Vercel (prod only)
        if: github.ref == 'refs/heads/main'
        run: npx vercel --prod --token=${{ secrets.VERCEL_TOKEN }}
🔒
Security
Always HTTPS. Set CSP headers. Store secrets in env vars — never hardcode in source code.
⚡
Performance
Lazy load images. Code-split routes. Use Next.js Image component for automatic optimization.
🔄
CI/CD
Automate tests and deploys. Use preview URLs for every PR. Never deploy to production manually.
📊
Monitoring
Add Sentry for error tracking. Monitor Core Web Vitals. Set alerts for downtime and P95 latency.
